// lib/ab-testing/framework.ts
export class ABTestingFramework {
  async createExperiment(
    name: string,
    variants: Variant[],
    trafficSplit: number[]
  ): Promise<Experiment> {
    return prisma.experiment.create({
      data: {
        name,
        status: 'running',
        variants: {
          create: variants.map((v, i) => ({
            name: v.name,
            config: v.config,
            trafficAllocation: trafficSplit[i],
          })),
        },
      },
      include: { variants: true },
    });
  }
  
  async getVariant(
    experimentName: string,
    userId: string
  ): Promise<string> {
    // Deterministic assignment based on user ID
    const hash = this.hashUserId(userId + experimentName);
    const experiment = await this.getExperiment(experimentName);
    
    let cumulative = 0;
    for (const variant of experiment.variants) {
      cumulative += variant.trafficAllocation;
      if (hash < cumulative) {
        return variant.name;
      }
    }
    
    return experiment.variants[0].name;
  }
  
  async trackConversion(
    experimentId: string,
    variantName: string,
    userId: string,
    value?: number
  ): Promise<void> {
    await prisma.experimentEvent.create({
      data: {
        experimentId,
        variantName,
        userId,
        eventType: 'conversion',
        value,
      },
    });
  }
  
  async getResults(experimentId: string): Promise<ExperimentResults> {
    const events = await prisma.experimentEvent.groupBy({
      by: ['variantName'],
      where: { experimentId, eventType: 'conversion' },
      _count: { id: true },
      _sum: { value: true },
    });
    
    const variants = experiment.variants.map(v => {
      const data = events.find(e => e.variantName === v.name);
      return {
        name: v.name,
        conversions: data?._count?.id || 0,
        revenue: data?._sum?.value || 0,
        conversionRate: 0, // calculate from traffic
      };
    });
    
    return { variants, significance: this.calculateSignificance(variants) };
  }
}

// lib/payments/processor.ts
export class PaymentProcessor {
  private stripe: Stripe;
  
  constructor() {
    this.stripe = new Stripe(process.env.STRIPE_SECRET_KEY!);
  }
  
  async createSubscription(
    userId: string,
    priceId: string,
    paymentMethodId: string
  ): Promise<Subscription> {
    const user = await prisma.user.findUnique({ where: { id: userId } });
    
    // Criar customer se n√£o existir
    let customerId = user.stripeCustomerId;
    if (!customerId) {
      const customer = await this.stripe.customers.create({
        email: user.email,
        name: `${user.firstName} ${user.lastName}`,
        payment_method: paymentMethodId,
        invoice_settings: { default_payment_method: paymentMethodId },
      });
      customerId = customer.id;
      
      await prisma.user.update({
        where: { id: userId },
        data: { stripeCustomerId: customerId },
      });
    }
    
    // Criar subscription
    const subscription = await this.stripe.subscriptions.create({
      customer: customerId,
      items: [{ price: priceId }],
      expand: ['latest_invoice.payment_intent'],
    });
    
    // Salvar no banco
    const dbSubscription = await prisma.subscription.create({
      data: {
        userId,
        stripeSubscriptionId: subscription.id,
        stripeCustomerId: customerId,
        status: subscription.status,
        currentPeriodStart: new Date(subscription.current_period_start * 1000),
        currentPeriodEnd: new Date(subscription.current_period_end * 1000),
        priceId,
      },
    });
    
    return dbSubscription;
  }
  
  async handleWebhook(event: Stripe.Event): Promise<void> {
    switch (event.type) {
      case 'invoice.payment_succeeded':
        await this.handlePaymentSucceeded(event.data.object);
        break;
        
      case 'invoice.payment_failed':
        await this.handlePaymentFailed(event.data.object);
        break;
        
      case 'customer.subscription.deleted':
        await this.handleSubscriptionCancelled(event.data.object);
        break;
        
      case 'customer.subscription.updated':
        await this.handleSubscriptionUpdated(event.data.object);
        break;
    }
  }
  
  private async handlePaymentSucceeded(invoice: Stripe.Invoice) {
    const subscription = await prisma.subscription.findFirst({
      where: { stripeSubscriptionId: invoice.subscription as string },
    });
    
    if (subscription) {
      await prisma.subscription.update({
        where: { id: subscription.id },
        data: { status: 'active' },
      });
      
      // Award points for payment
      await awardPoints(subscription.userId, 'subscription_payment', 500);
    }
  }
}

  private async calculateMRR(timeRange: TimeRange): Promise<number> {
    const activeSubscriptions = await prisma.subscription.findMany({
      where: {
        status: 'active',
        createdAt: { gte: timeRange.start, lte: timeRange.end },
      },
      include: { price: true },
    });
    
    return activeSubscriptions.reduce((total, sub) => {
      return total + (sub.price?.amount || 0);
    }, 0);
  }
  
  private async calculateChurnRate(timeRange: TimeRange): Promise<number> {
    const startDate = timeRange.start;
    const endDate = timeRange.end;
    
    const customersAtStart = await prisma.subscription.count({
      where: {
        status: 'active',
        createdAt: { lte: startDate },
      },
    });
    
    const churnedCustomers = await prisma.subscription.count({
      where: {
        status: 'cancelled',
        cancelledAt: { gte: startDate, lte: endDate },
      },
    });
    
    return (churnedCustomers / customersAtStart) * 100;
  }
}
